<script>
const articles = [
  {title:"What is Nutrition?", content:"Nutrition is the study of nutrients in food..."},
  {title:"Food Safety Basics", content:"Food safety prevents contamination..."},
  {title:"Food Chemistry", content:"Food chemistry explains reactions in food..."}
];

const grid = document.getElementById('article-grid');

function renderArticles(list){
  grid.innerHTML='';
  list.forEach((a,i)=>{
    grid.innerHTML += `<div class="article" onclick="openModal(${i})">
      <h3>${a.title}</h3>
      <p>${a.content.substring(0,50)}...</p>
      <button>Read More</button>
    </div>`;
  });
}

function openModal(i){
  document.getElementById('modal-title').innerText = articles[i].title;
  document.getElementById('modal-content-text').innerText = articles[i].content;
  document.getElementById('modal').style.display='block';
}

document.getElementById('close').onclick = ()=>{
  document.getElementById('modal').style.display='none';
};

// Search

document.getElementById('search').addEventListener('input', e=>{
  const value = e.target.value.toLowerCase();
  const filtered = articles.filter(a=>a.title.toLowerCase().includes(value));
  renderArticles(filtered);
});

// Form validation

document.getElementById('contact-form').addEventListener('submit', e=>{
  e.preventDefault();
  alert('Message sent successfully!');
});

renderArticles(articles);
</script>
